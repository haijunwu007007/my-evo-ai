@app.get("/apps")
async def apps_list():
    apps_dir = BASE_DIR / "output" / "apps"
    if not apps_dir.exists(): return JSONResponse({"apps":[]})
    apps = []
    for f in sorted(apps_dir.glob("*.html"), reverse=True)[:50]:
        size = f.stat().st_size
        apps.append({"name":f.stem, "url":f"/output/apps/{f.name}", "size":size, "date":time.ctime(f.stat().st_mtime)})
    html = """<!DOCTYPE html><html><head><meta charset="UTF-8"><title>已生成的APP</title><meta name="viewport" content="width=device-width,initial-scale=1"><style>body{font-family:-apple-system,system-ui,sans-serif;background:#0f0f1a;color:#e2e8f0;max-width:800px;margin:0 auto;padding:20px}h1{font-size:24px;margin-bottom:20px}.app{background:#1a1a2e;border-radius:12px;padding:16px;margin-bottom:12px;border:1px solid #2d2d4a}.app:hover{border-color:#6366f1}.app a{color:#818cf8;text-decoration:none;font-size:16px;display:block}.app .meta{color:#64748b;font-size:12px;margin-top:4px}.app .size{color:#22c55e}.empty{text-align:center;padding:60px 20px;color:#64748b}</style></head><body><h1>📂 已生成的 APP</h1>"""
    if not apps: html += '<div class="empty">还没有生成过APP<br>试试说"开发一个任务管理系统"</div>'
    for a in apps:
        html += f'<div class="app"><a href="{a["url"]}" target="_blank">{a["name"][:40]}</a><div class="meta"><span class="size">{(a["size"]/1024):.1f}KB</span> · {a["date"]}</div></div>'
    html += "</body></html>"
    return HTMLResponse(html)
